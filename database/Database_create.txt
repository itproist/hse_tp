CREATE TYPE quisine AS ENUM ('Russian', 'Italian', 'Spanish' ,'Japanese', 'Chinese');
CREATE TABLE Dishes
	(dish_id BIGINT NOT NULL PRIMARY KEY,
	dish_name text NOT NULL,
	time interval NOT NULL,
	description text NOT NULL,
	dish_quisine quisine NOT NULL,
	recepy text NOT NULL);

CREATE TYPE measure AS ENUM ('milliliters', 'grams');
CREATE TABLE Products 
	(product_id BIGINT NOT NULL PRIMARY KEY,
	name text NOT NULL,
	measure_type measure NOT NULL);


CREATE TABLE Ingredients
	(ingredient_id BIGINT NOT NULL PRIMARY KEY,
	dish_id BIGINT NOT NULL,
	product_id BIGINT NOT NULL,
	amount BIGINT NOT NULL,

	CONSTRAINT dish_id_fk FOREIGN KEY (dish_id) REFERENCES dishes
	CONSTRAINT product_id_fk FOREIGN KEY (product_id) REFERENCES products)
